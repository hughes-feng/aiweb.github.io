    <script>
        // 1. Lenis 平滑滚动
        const lenis = new Lenis({
            duration: 1.2,
            easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)), 
            smoothWheel: true
        });
        function raf(time) { lenis.raf(time); requestAnimationFrame(raf); }
        requestAnimationFrame(raf);

        // 2. 导航栏滚动变化 & 移动端菜单
        const navbar = document.getElementById('navbar');
        window.addEventListener('scroll', () => {
            if (window.scrollY > 50) navbar.classList.add('scrolled');
            else navbar.classList.remove('scrolled');
        });
        const menuBtn = document.getElementById('menuBtn');
        const navLinks = document.getElementById('navLinks');
        menuBtn.addEventListener('click', () => {
            navLinks.classList.toggle('hidden');
            navLinks.classList.toggle('flex');
            navLinks.classList.toggle('active');
        });

        // 3. GSAP 滚动渐入动画
        gsap.registerPlugin(ScrollTrigger);
        document.querySelectorAll('.gs_reveal').forEach((elem) => {
            gsap.to(elem, {
                scrollTrigger: { trigger: elem, start: "top 85%", toggleClass: "active", once: true }
            });
        });

        // 4. 自定义光标跟随 (移动端自动忽略)
        const cursorCore = document.getElementById('cursorCore');
        const cursorGlow = document.getElementById('cursorGlow');
        let mouseX = window.innerWidth/2, mouseY = window.innerHeight/2;
        let cX = mouseX, cY = mouseY;

        window.addEventListener('mousemove', (e) => {
            mouseX = e.clientX; mouseY = e.clientY;
        });
        
        gsap.ticker.add(() => {
            cX += (mouseX - cX) * 0.2; cY += (mouseY - cY) * 0.2;
            cursorCore.style.transform = `translate(calc(-50% + ${cX}px), calc(-50% + ${cY}px))`;
            cursorGlow.style.transform = `translate(calc(-50% + ${cX}px), calc(-50% + ${cY}px))`;
        });

        // 绑定光标悬停交互
        document.querySelectorAll('.interactable, a, button').forEach(el => {
            el.addEventListener('mouseenter', () => document.body.classList.add('is-hover'));
            el.addEventListener('mouseleave', () => document.body.classList.remove('is-hover'));
        });

        // 5. 卡片鼠标悬浮发光效果
        document.querySelectorAll(".card").forEach(card => {
            card.addEventListener('mousemove', e => {
                const rect = card.getBoundingClientRect();
                card.style.setProperty("--mouse-x", `${e.clientX - rect.left}px`);
                card.style.setProperty("--mouse-y", `${e.clientY - rect.top}px`);
            });
        });

        // 6. 爱心点赞与撒花特效
        function doLike(element, event) {
            element.classList.toggle('active');
            if(element.classList.contains('active')) {
                element.innerHTML = '♥';
                // 撒花位置计算
                const rect = element.getBoundingClientRect();
                const x = (rect.left + rect.width/2) / window.innerWidth;
                const y = (rect.top + rect.height/2) / window.innerHeight;
                confetti({
                    particleCount: 50, spread: 60, origin: { x: x, y: y },
                    colors: ['#ef4444', '#f43f5e', '#ffffff'], disableForReducedMotion: true
                });
            } else {
                element.innerHTML = '♡';
            }
        }

        // 7. 弹窗控制逻辑
        function openMemberModal(name, role, desc, avatar) {
            document.getElementById('memberModalName').innerText = name;
            document.getElementById('memberModalRole').innerText = role;
            document.getElementById('memberModalDesc').innerText = desc;
            document.getElementById('memberModalAvatar').src = avatar;
            document.getElementById('memberModal').classList.add('active');
        }

        function openVideoModal(title, subtitle, desc, vidSrc) {
            document.getElementById('videoModalTitle').innerText = title;
            document.getElementById('videoModalSubtitle').innerText = subtitle;
            document.getElementById('videoModalDesc').innerText = desc;
            
            const video = document.getElementById('modalVideo');
            video.src = vidSrc;
            document.getElementById('videoModal').classList.add('active');
            video.play().catch(e => console.log("Auto-play blocked by browser."));
        }

        function showTip(title, message) {
            document.getElementById('tipModalTitle').innerText = title;
            document.getElementById('tipModalDesc').innerText = message;
            document.getElementById('tipModal').classList.add('active');
        }

        function closeModal(id) {
            document.getElementById(id).classList.remove('active');
            // 如果关闭的是视频弹窗，暂停视频
            if(id === 'videoModal') {
                const video = document.getElementById('modalVideo');
                video.pause();
                video.currentTime = 0;
            }
        }

        // 8. 上传表单文件选择与预览验证
        function handleFileSelect(e) {
            const file = e.target.files[0];
            if(!file) return;
            const imgPreview = document.getElementById('imgPreview');
            const videoPreview = document.getElementById('videoPreview');
            imgPreview.style.display = 'none'; videoPreview.style.display = 'none';

            if(file.type.startsWith('image/')) {
                imgPreview.src = URL.createObjectURL(file);
                imgPreview.style.display = 'block';
            } else if(file.type.startsWith('video/')) {
                videoPreview.src = URL.createObjectURL(file);
                videoPreview.style.display = 'block';
            }
        }

        function resetPreviews() {
            document.getElementById('imgPreview').style.display = 'none';
            document.getElementById('videoPreview').style.display = 'none';
        }

        function handleFormSubmit(e) {
            e.preventDefault();
            let isValid = true;
            const user = document.getElementById('uploadUser');
            const desc = document.getElementById('uploadDesc');
            
            // 简单校验
            if(!user.value.trim()) { user.classList.add('error'); document.getElementById('userTip').style.display='block'; isValid=false; }
            else { user.classList.remove('error'); document.getElementById('userTip').style.display='none'; }

            if(!desc.value.trim()) { desc.classList.add('error'); document.getElementById('descTip').style.display='block'; isValid=false; }
            else { desc.classList.remove('error'); document.getElementById('descTip').style.display='none'; }

            if(isValid) {
                showTip("提交成功", "您的作品已成功上传，我们会尽快审核处理！");
                e.target.reset();
                resetPreviews();
            }
        }
    </script>
